// CONTACT FORM document.getElementById('contactForm').addEventListener('submit', async function(e) { e.preventDefault(); const form = e.target; const btn = document.getElementById('submitBtn'); const successMsg = document.getElementById('form-success'); const errorMsg = document.getElementById('form-error'); btn.innerHTML = 'Sending...'; btn.disabled = true; successMsg.style.display = 'none'; errorMsg.style.display = 'none'; try { const response = await fetch('https://formspree.io/f/xkoqkydz', { method: 'POST', body: new FormData(form), headers: { 'Accept': 'application/json' } }); if (response.ok) { form.reset(); successMsg.style.display = 'block'; btn.innerHTML = 'Send Message'; btn.disabled = false; } else { errorMsg.style.display = 'block'; btn.innerHTML = 'Send Message'; btn.disabled = false; } } catch (error) { errorMsg.style.display = 'block'; btn.innerHTML = 'Send Message'; btn.disabled = false; } }); // ACTIVE NAV HIGHLIGHT const allSections = document.querySelectorAll('section[id], div[id]'); const navAnchors = document.querySelectorAll('.nav-links a'); window.addEventListener('scroll', () => { let current = ''; allSections.forEach(s => { if (window.scrollY >= s.offsetTop - 200) current = s.getAttribute('id'); }); navAnchors.forEach(a => { a.style.color = ''; if (a.getAttribute('href') === '#' + current) a.style.color = 'var(--primary)'; }); });
